def _refresh_colors(self):
    """Обновляет только цвета маркеров — координаты не трогает."""
    if not self.project.is_open or not self.db:
        QMessageBox.warning(self, "Проект не загружен",
                            "Сначала откройте проект.")
        return
    # Только цвета — без перезагрузки позиций
    sensors = self.db.get_all_sensors()
    self.scene.update_marker_colors(sensors)
    # Реестр — только обновляем подсветку, не перезагружаем модель
    if self.registry_window and self.registry_window.isVisible():
        self.registry_window.update_schema_tags(
            set(self.scene._markers.keys())
        )
    self.statusBar().showMessage("Цвета маркеров обновлены")

def add_empty_row(self):
    pos = len(self._rows)
    db_cols_count = len(self._headers) - 2  # без "№" и вычисляемого

    # Следующий порядковый номер = максимальный rowid + 1
    next_rowid = max(self._rowids, default=0) + 1

    self.beginInsertRows(QModelIndex(), pos, pos)
    self._rows.append([""] * db_cols_count)
    self._rowids.append(next_rowid)
    self._new_row_indices.add(pos)
    self.endInsertRows()

        self.model.modelReset.connect(self._update_count)

    def _find_on_schema(self):
        """Отправляет TAG выбранной строки в MainWindow для поиска на схеме."""
        idx = self.table_view.currentIndex()
        if not idx.isValid():
            QMessageBox.information(
                self, "Выберите строку",
                "Выберите датчик в таблице для поиска на схеме."
            )
            return
        tag = self.model._get_tag(idx.row())
        if not tag or tag == "—":
            QMessageBox.information(
                self, "Нет TAG",
                "У выбранной строки не заполнен код параметра (TAG)."
            )
            return
        if tag not in self.model._tags_on_schema:
            QMessageBox.information(
                self, "Не на схеме",
                f"Датчик «{tag}» не размещён на схеме."
            )
            return
        self.find_on_schema.emit(tag)

def _open_registry(self):
    if not self.project.is_open or not self.db:
        QMessageBox.warning(self, "Проект не загружен",
                            "Сначала создайте или откройте проект.")
        return
    schema_tags = set(self.scene._markers.keys())
    if self.registry_window is None:
        self.registry_window = RegistryWindow(self.db.conn, schema_tags)
        # Подключаем сигнал поиска на схеме
        self.registry_window.find_on_schema.connect(self._find_sensor_by_tag)
    else:
        self.registry_window.update_schema_tags(schema_tags)
    self.registry_window.show()
    self.registry_window.raise_()

def _find_sensor_by_tag(self, tag: str):
    """Центрирует схему на датчике с указанным TAG."""
    marker = self.scene._markers.get(tag)
    if not marker:
        return
    from core.sensor_marker import SensorMarker
    target = SensorMarker.RADIUS * 2 * 8
    rect   = QRectF(
        marker.x() - target / 2,
        marker.y() - target / 2,
        target, target
    )
    self.view.fitInView(rect, Qt.KeepAspectRatio)
    # Поднимаем главное окно на передний план
    self.raise_()
    self.activateWindow()
    self.statusBar().showMessage(f"Найден на схеме: {tag}")
